MergeSort

Definition

Mergesort is a stable sort algorithm and one of the fastest sorting algorithm. The technique used in the sorting algorithm is so called divide and conquer (which basically means divide the bigger problems into smaller solvable chucnks and solve those smaller chunks and combine them to full solution). Mergesort was origianlly invented by one of the early computer pioneer Jon Von Neuwmann who also happen make the theoritical model of Universal Turing Machine into practice by inventing Stored Program Comouter.

sort(arr[], aux[], low, hi):
  // if hi is less than equal to low we just return
  if (hi <= low) return
  //get the mid index of the container(array)
  mid = low + (hi - low) / 2
  //Recursively call the first half of the array
  sort(arr, aux, low, mid)
  //Recursively call the right half of the array
  sort(arr, aux, mid + 1, hi)
  //merge the succesive elements in sorted order by calling another subroutine merge
  merge(arr, aux, low, mid, hi):

merge(arr[], aux[], low, mid, high):
  //Copy the elements of arr[] into an auxilliary array
  for k = low; k <= high; k++:
    aux[k] = arr[k]
    
  // left part of the first half of the array
  i = low
  // left part of the second half of the array
  j = mid + 1;
​
  //Iteratively compare the array elements and put them in right order
  for k = low; k <= high; k++:
  //left half of the array has been exhausted. In that case,move go to the second half and get the element and      shift the index.
    if (i > mid):
      arr[k] = aux[j++]
    // Right half of the array is exhaustes,move to the first half and get the element and shift the index.
    else if (j > high):
      arr[k] = aux[i++]
    //Compare the element of the two subarrays 
    else if arr[j] < arr[i]:
      //This means the element in the right is greater than on the left.
      arr[k] = aux[j++]
    // There is nothing to compare just get on with left part of the array
    else:
      arr[k] = arr[i++]

public class MergeSort {
  public static <T extends Comparable<T>> void sort(T[] arr) {
    T[] aux = (T[]) new Comparable[arr.length];
    sort(arr, aux, 0, arr.length - 1);
  }
  
  private static <T extends Comparable<T>> void sort(T[] arr, T[] aux, int low, int high) {
    if (high <= low) return;
    int mid = low + (high - low) / 2;
    sort(arr, aux, low, mid);
    sort(arr, aux, mid + 1, high);
    merge(arr, aux, low, mid, high);
  }
  
  private static <T extends Comparable<T>> void merge(T[] arr, T[] aux, int low, int mid, int high) {
    for (int i = low; i <= high; i++) {
      aux[i] = arr[i];
    }
    int i = low, j = mid + 1;
    for (int k = low; k <= high; k++) {
      if (i > mid)                    arr[k] = aux[j++];
      else if (j > high)              arr[k] = aux[i++];
      else if (less(aux[j], aux[i]))  arr[k] = aux[j++];
      else                            arr[k] = aux[i++];
    }
  }
  private static <T extends Comparable<T>> boolean less(T a, T b) {
    return a.compareTo(b) < 0;
  }
}
